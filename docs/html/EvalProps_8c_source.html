26void EvalProps() {
27 real v, vv;
28 virSum = 0.0;
29 vSumX = 0.0; vSumY = 0.0; vSum = 0.0;
30 vvSum = 0.;
31 int n;
32
33 for (n = 1; n <= nAtom; n++) {
34 vv = 0.;
35 // Initialize v with a default value to avoid "uninitialized" warning.
36 v = 0.0;
37 // X direction velocity
38 if (strcmp(solver, "Verlet") == 0) {
39 v = vx[n];
40 } else if (strcmp(solver, "LeapFrog") == 0) {
41 v = vx[n] - 0.5 * deltaT * ax[n];
42 }
43 vSum += v;
44 vv += Sqr(v);
45 vSumX += v;
46 // Y direction velocity
47 if (strcmp(solver, "Verlet") == 0) {
48 v = vy[n];
49 } else if (strcmp(solver, "LeapFrog") == 0) {
50 v = vy[n] - 0.5 * deltaT * ay[n];
51 }
52 vSum += v;
53 vSumY += v;
54 vv += Sqr(v);
55 vvSum += vv;
56 }
57
58 kinEnergy = 0.5 * vvSum / nAtom ;
59 uSumPairPerAtom = uSumPair / nAtom ;
60 BondEnergyPerAtom = TotalBondEnergy / (0.5*nAtom); //Factor of 0.5 since each atom has one half the bond energy
61 potEnergy = uSumPairPerAtom + BondEnergyPerAtom ;
62 totEnergy = kinEnergy + potEnergy;
63 virSumxx = virSumPairxx + virSumBondxx ;
64 virSumyy = virSumPairyy + virSumBondyy ;
65 virSumxy = virSumPairxy + virSumBondxy ;
66 virSum = virSumPair + virSumBond;
67 pressure = density * (vvSum + virSum) / (nAtom * NDIM);
68
69}
