26void VelocityVerletStep(int icode){
27int n;
28 if(icode == 1){
29 for (n= 1; n <= nAtom; n++) {
30 if(atomType[n] != freezeAtomType){
31 vx[n] += ax[n] * 0.5 * deltaT;
32 vy[n] += ay[n] * 0.5 * deltaT;
33 rx[n] += vx[n] * deltaT;
34 ry[n] += vy[n] * deltaT;
35 }
36 //Calculating the image flags here
37 if (rx[n] >= regionH[1]) {
38 rx[n] -= region[1];
39 ImageX[n]++;
40 } else if (rx[n] < -regionH[1]) {
41 rx[n] += region[1];
42 ImageX[n]--;
43 }
44 if (ry[n] >= regionH[2]) {
45 ry[n] -= region[2];
46 ImageY[n]++;
47 } else if (ry[n] < -regionH[2]) {
48 ry[n] += region[2];
49 ImageY[n]--;
50 } } }
51 else if(icode == 2){
52 for(n = 1; n <= nAtom; n++) {
53 if(atomType[n] != freezeAtomType){
54 vx[n] += ax[n] * 0.5 * deltaT;
55 vy[n] += ay[n] * 0.5 * deltaT;
56} } } }
